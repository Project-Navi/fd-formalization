/-!
# Flower Graph Exact Counts

Exact formulas for the edge and vertex counts of (u,v)-flower graphs
at generation `g`. With `w = u + v`:

- **Edges:** `E_g = w^g`
- **Vertices:** `(w - 1) * N_g = (w - 2) * w^g + w`

These are proved by induction on `g` from the construction rule:
each edge is replaced by two parallel paths of lengths `u` and `v`,
contributing `(u - 1) + (v - 1) = w - 2` new internal vertices per
edge and multiplying the edge count by `w`.

The counting formulas are pure arithmetic recurrences and do not
depend on the graph representation.

## Main definitions

- `flowerEdgeCount` ‚Äî number of edges at generation `g`
- `flowerVertCount` ‚Äî number of vertices at generation `g`
- `flowerVertCountReal` ‚Äî vertex count cast to `‚Ñù`

## Main statements

- `flowerEdgeCount_eq_pow` ‚Äî `E_g = w^g`
- `flowerVertCount_eq` ‚Äî `(w - 1) * N_g = (w - 2) * w^g + w`
- `flowerVertCount_lower` ‚Äî `(w - 2) * w^g ‚â§ (w - 1) * N_g`
- `flowerVertCount_upper` ‚Äî `(w - 1) * N_g ‚â§ 2 * (w - 1) * w^g`
- `flowerVertCount_pos` ‚Äî `0 < N_g`

## References

- [Rozenfeld2007] ¬ß2, construction and counting formulas.

## Tags

flower graph, fractal, edge count, vertex count
-/

/-- Number of edges in the (u,v)-flower at generation `g`. -/
def flowerEdgeCount (u v : ‚Ñï) : ‚Ñï ‚Üí ‚Ñï
  | 0 => 1
  | g + 1 => (u + v) * flowerEdgeCount u v g

@[simp]
theorem flowerEdgeCount_zero (u v : ‚Ñï) : flowerEdgeCount u v 0 = 1 := rfl

@[simp]
theorem flowerEdgeCount_succ (u v g : ‚Ñï) :
    flowerEdgeCount u v (g + 1) = (u + v) * flowerEdgeCount u v g := rfl

/-- Number of vertices in the (u,v)-flower at generation `g`. -/
def flowerVertCount (u v : ‚Ñï) : ‚Ñï ‚Üí ‚Ñï
  | 0 => 2
  | g + 1 => flowerVertCount u v g + (u + v - 2) * flowerEdgeCount u v g

@[simp]
theorem flowerVertCount_zero (u v : ‚Ñï) : flowerVertCount u v 0 = 2 := rfl

@[simp]
theorem flowerVertCount_succ (u v g : ‚Ñï) :
    flowerVertCount u v (g + 1) =
    flowerVertCount u v g + (u + v - 2) * flowerEdgeCount u v g := rfl

/-! ### Edge count -/

/-- The edge count of the (u,v)-flower at generation `g` is `(u + v) ^ g`. -/
theorem flowerEdgeCount_eq_pow (u v g : ‚Ñï) :
    flowerEdgeCount u v g = (u + v) ^ g := by
  induction g with
  | zero => simp
  | succ g ih => rw [flowerEdgeCount_succ, ih, pow_succ, mul_comm]

/-! ### Vertex count -/

/-- The vertex count satisfies `(w - 1) * N_g = (w - 2) * w^g + w`
where `w = u + v`. Multiplicative form avoids ‚Ñï division. -/
theorem flowerVertCount_eq (u v g : ‚Ñï) (hu : 1 < u) (huv : u ‚â§ v) :
    (u + v - 1) * flowerVertCount u v g =
    (u + v - 2) * (u + v) ^ g + (u + v) := by
  induction g with
  | zero =>
    simp
    omega
  | succ g ih =>
    simp only [flowerVertCount_succ, flowerEdgeCount_eq_pow, pow_succ]
    have h1 : 1 ‚â§ u + v := by omega
    have h2 : 2 ‚â§ u + v := by omega
    zify [h1, h2] at ih ‚ä¢
    nlinarith [ih]

/-- The vertex count is always positive (holds for all `u`, `v`). -/
theorem flowerVertCount_pos (u v g : ‚Ñï) :
    0 < flowerVertCount u v g := by
  induction g with
  | zero => simp
  | succ g ih =>
    simp only [flowerVertCount_succ]
    omega

/-! ### Two-sided bounds for squeeze -/

/-- Lower bound: `(w - 2) * w^g ‚â§ (w - 1) * N_g`. -/
theorem flowerVertCount_lower (u v g : ‚Ñï) (hu : 1 < u) (huv : u ‚â§ v) :
    (u + v - 2) * (u + v) ^ g ‚â§
    (u + v - 1) * flowerVertCount u v g := by
  rw [flowerVertCount_eq u v g hu huv]
  exact le_add_of_nonneg_right (Nat.zero_le _)

/-- Upper bound: `(w - 1) * N_g ‚â§ 2 * (w - 1) * w^g`. -/
theorem flowerVertCount_upper (u v g : ‚Ñï) (hu : 1 < u) (huv : u ‚â§ v) :
    (u + v - 1) * flowerVertCount u v g ‚â§
    2 * (u + v - 1) * (u + v) ^ g := by
  rw [flowerVertCount_eq u v g hu huv]
  have h1 : 1 ‚â§ u + v := by omega
  have h2 : 2 ‚â§ u + v := by omega
  have h_pow : 1 ‚â§ (u + v) ^ g := Nat.one_le_pow _ _ (by omega)
  zify [h1, h2] at h_pow ‚ä¢
  nlinarith [h_pow]

/-! ### Monotonicity -/

/-- Edge count is positive when `u + v > 0`. -/
theorem flowerEdgeCount_pos (u v g : ‚Ñï) (hw : 0 < u + v) :
    0 < flowerEdgeCount u v g := by
  exact Nat.recOn g ( Nat.succ_pos _ ) fun n ih => mul_pos hw ih;

/-- Edge count grows strictly between generations. -/
theorem flowerEdgeCount_strict_mono (u v g : ‚Ñï) (hw : 1 < u + v) :
    flowerEdgeCount u v g < flowerEdgeCount u v (g + 1) := by
  exact lt_mul_of_one_lt_left ( Nat.recOn g ( by tauto ) fun n ihn => by rw [ flowerEdgeCount_succ ] ; nlinarith ) ( by linarith )

/-- Vertex count grows strictly between generations. -/
theorem flowerVertCount_strict_mono (u v g : ‚Ñï) (hu : 1 < u) (huv : u ‚â§ v) :
    flowerVertCount u v g < flowerVertCount u v (g + 1) := by
  -- By definition of $flowerVertCount$, we have $flowerVertCount u v (g + 1) = flowerVertCount u v g + (u + v - 2) * flowerEdgeCount u v g$.
  have h_def : flowerVertCount u v (g + 1) = flowerVertCount u v g + (u + v - 2) * flowerEdgeCount u v g := by
    rfl;
  exact h_def.symm ‚ñ∏ lt_add_of_pos_right _ ( Nat.mul_pos ( Nat.sub_pos_of_lt ( by linarith ) ) ( flowerEdgeCount_pos u v g ( by linarith ) ) )

/-! ### Cast identity -/

/-- Vertex count recurrence in `‚Ñù`:
`(w - 1) * N_g = (w - 2) * w^g + w` where `w = u + v`. -/
theorem flowerVertCount_cast_eq (u v g : ‚Ñï) (hu : 1 < u) (huv : u ‚â§ v) :
    (‚Üë(u + v) - 1 : ‚Ñù) * (‚Üë(flowerVertCount u v g) : ‚Ñù) =
    (‚Üë(u + v) - 2 : ‚Ñù) * (‚Üë(u + v) : ‚Ñù) ^ g + (‚Üë(u + v) : ‚Ñù) := by
  convert congr_arg ( fun x : ‚Ñï => x : ‚Ñï ‚Üí ‚Ñù ) ( flowerVertCount_eq u v g hu huv ) using 1 ; ring_nf; norm_cast;
  ¬∑ rw [ Int.subNatNat_eq_coe ] ; push_cast ; ring;
    rw [ Nat.cast_sub ] <;> push_cast <;> linarith;
  ¬∑ simp +zetaDelta at *;
    exact Or.inl ( by rw [ Nat.cast_sub ] <;> push_cast <;> linarith )

/-! ### Real-valued wrappers -/

/-- Vertex count cast to `‚Ñù`. -/
noncomputable def flowerVertCountReal (u v : ‚Ñï) (g : ‚Ñï) : ‚Ñù :=
  (flowerVertCount u v g : ‚Ñù)

/-- The real-valued vertex count is positive. -/
theorem flowerVertCountReal_pos (u v g : ‚Ñï) :
    0 < flowerVertCountReal u v g := by
  simp only [flowerVertCountReal]
  exact Nat.cast_pos.mpr (flowerVertCount_pos u v g)
